## ALGORITMO DE PUNTUACIÓN DE CAOS

### Concepto
Mide qué tan incoherente es la historia detectando cambios bruscos de tema mediante análisis de palabras clave por género.

### Implementación (utils/caos.js)

```javascript
// === src/utils/caos.js ===

// Palabras clave por género para detección de temas
const PALABRAS_CLAVE = {
  terror: ['muerte', 'sangre', 'oscuro', 'miedo', 'fantasma', 'terror', 'grito', 'sombra', 'noche', 'huir'],
  romance: ['amor', 'corazón', 'beso', 'mirada', 'sentir', 'abrazar', 'juntos', 'siempre', 'alma', 'destino'],
  aventura: ['tesoro', 'mapa', 'peligro', 'descubrir', 'explorar', 'viaje', 'cruzar', 'misterio', 'buscar', 'encontrar'],
  sciFi: ['nave', 'tecnología', 'futuro', 'robot', 'espacio', 'tiempo', 'dimensional', 'científico', 'planeta', 'alien'],
  fairyTale: ['hada', 'príncipe', 'princesa', 'mágico', 'encantado', 'castillo', 'dragón', 'varita', 'hechizo', 'reino'],
  comedia: ['risa', 'divertido', 'absurdo', 'ridículo', 'cómico', 'gracioso', 'payaso', 'broma', 'humor', 'carcajada'],
  misterio: ['pista', 'sospechoso', 'detective', 'resolver', 'enigma', 'secreto', 'oculto', 'descubrir', 'investigar', 'crimen'],
  western: ['pistola', 'caballo', 'sheriff', 'saloon', 'desierto', 'forajido', 'duelo', 'frontera', 'revólver', 'galope']
};

/**
 * Calcula el score de caos de una historia
 * @param {Array} contribuciones - Array de objetos {jugador, texto, ronda}
 * @param {String} generoBase - Género original de la historia
 * @returns {Number} Score de 0 (coherente) a 100 (caos total)
 */
export function calcularCaos(contribuciones, generoBase) {
  if (contribuciones.length < 2) return 0;

  let transiciones = 0;
  let temaActual = generoBase;
  
  contribuciones.forEach((contribucion, index) => {
    if (index === 0) return; // Skip primera contribución
    
    const texto = contribucion.texto.toLowerCase();
    const temasDetectados = detectarTemas(texto);
    
    // Si el tema más fuerte es diferente al actual, hay transición
    if (temasDetectados.length > 0 && temasDetectados[0] !== temaActual) {
      transiciones++;
      temaActual = temasDetectados[0];
    }
    
    // Bonus de caos si mezcla múltiples géneros en una misma oración
    if (temasDetectados.length > 2) {
      transiciones += 0.5;
    }
  });
  
  // Normalizar a escala 0-100
  // Asumiendo que 5+ transiciones = caos máximo
  const maxTransiciones = contribuciones.length - 1;
  const caosNormalizado = Math.min((transiciones / maxTransiciones) * 100, 100);
  
  return Math.round(caosNormalizado);
}

/**
 * Detecta qué temas/géneros están presentes en un texto
 * @param {String} texto - Texto a analizar
 * @returns {Array} Géneros detectados ordenados por frecuencia
 */
function detectarTemas(texto) {
  const conteoGeneros = {};
  
  Object.keys(PALABRAS_CLAVE).forEach(genero => {
    const palabras = PALABRAS_CLAVE[genero];
    let coincidencias = 0;
    
    palabras.forEach(palabra => {
      // Usar regex para buscar palabras completas
      const regex = new RegExp(`\\b${palabra}\\w*\\b`, 'gi');
      const matches = texto.match(regex);
      if (matches) coincidencias += matches.length;
    });
    
    if (coincidencias > 0) {
      conteoGeneros[genero] = coincidencias;
    }
  });
  
  // Ordenar géneros por número de coincidencias (descendente)
  return Object.entries(conteoGeneros)
    .sort((a, b) => b[1] - a[1])
    .map(entry => entry[0]);
}

/**
 * Genera descripción del nivel de caos
 * @param {Number} score - Score de 0-100
 * @returns {String} Descripción textual
 */
export function descripcionCaos(score) {
  if (score < 20) return "Historia sorprendentemente coherente";
  if (score < 40) return "Caos moderado, con sentido general";
  if (score < 60) return "Giros inesperados constantes";
  if (score < 80) return "Locura narrativa desatada";
  return "Caos absoluto y glorioso";
}
```

### Ejemplo de Cálculo

**Historia (3 jugadores, género Romance):**

1. Ana (Romance): "Sus miradas se cruzaron cuando ambos alcanzaron el último aguacate..."
   - Tema detectado: Romance (amor, mirada) → Sin transición
   
2. Luis (Terror): "Pero el aguacate sangraba y susurraba su nombre en la oscuridad..."
   - Temas detectados: Terror (sangre, oscuro, susurrar) → **Transición 1**
   
3. Carmen (Comedia): "Ana pensó 'definitivamente necesito cambiar de supermercado'..."
   - Temas detectados: Comedia (absurdo, ridículo) → **Transición 2**

**Resultado:** 2 transiciones en 2 contribuciones = 100% de cambios = **Score: 100**

// === src/utils/audio.js ===

let audioContext = null;

// Inicializar contexto de audio
export function initAudio() {
  if (!audioContext) {
    audioContext = new (window.AudioContext || window.webkitAudioContext)();
  }
  return audioContext;
}

/**
 * Sonido de tecla individual (máquina de escribir)
 * Frecuencia: 200-400 Hz con decay rápido
 */
export function playKeystroke() {
  const ctx = initAudio();
  const oscillator = ctx.createOscillator();
  const gainNode = ctx.createGain();
  
  oscillator.connect(gainNode);
  gainNode.connect(ctx.destination);
  
  // Frecuencia aleatoria para variación
  oscillator.frequency.value = 200 + Math.random() * 200;
  oscillator.type = 'square';
  
  // Envelope: ataque rápido, decay corto
  gainNode.gain.setValueAtTime(0.05, ctx.currentTime);
  gainNode.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.05);
  
  oscillator.start(ctx.currentTime);
  oscillator.stop(ctx.currentTime + 0.05);
}

/**
 * Sonido de campana al terminar oración
 * Frecuencia: 800 Hz con reverb simulado
 */
export function playBell() {
  const ctx = initAudio();
  const oscillator = ctx.createOscillator();
  const gainNode = ctx.createGain();
  
  oscillator.connect(gainNode);
  gainNode.connect(ctx.destination);
  
  oscillator.frequency.value = 800;
  oscillator.type = 'sine';
  
  gainNode.gain.setValueAtTime(0.1, ctx.currentTime);
  gainNode.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.5);
  
  oscillator.start(ctx.currentTime);
  oscillator.stop(ctx.currentTime + 0.5);
}

/**
 * Sonido de transición entre turnos
 * Whoosh descendente
 */
export function playTransition() {
  const ctx = initAudio();
  const oscillator = ctx.createOscillator();
  const gainNode = ctx.createGain();
  
  oscillator.connect(gainNode);
  gainNode.connect(ctx.destination);
  
  oscillator.frequency.setValueAtTime(600, ctx.currentTime);
  oscillator.frequency.exponentialRampToValueAtTime(200, ctx.currentTime + 0.3);
  oscillator.type = 'sawtooth';
  
  gainNode.gain.setValueAtTime(0.08, ctx.currentTime);
  gainNode.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.3);
  
  oscillator.start(ctx.currentTime);
  oscillator.stop(ctx.currentTime + 0.3);
}

/**
 * Fanfarria de revelación final
 * Secuencia de notas ascendentes
 */
export function playRevealFanfare() {
  const ctx = initAudio();
  const notes = [261.63, 329.63, 392.00, 523.25]; // C4, E4, G4, C5
  
  notes.forEach((freq, index) => {
    const oscillator = ctx.createOscillator();
    const gainNode = ctx.createGain();
    
    oscillator.connect(gainNode);
    gainNode.connect(ctx.destination);
    
    oscillator.frequency.value = freq;
    oscillator.type = 'triangle';
    
    const startTime = ctx.currentTime + (index * 0.15);
    gainNode.gain.setValueAtTime(0.1, startTime);
    gainNode.gain.exponentialRampToValueAtTime(0.01, startTime + 0.4);
    
    oscillator.start(startTime);
    oscillator.stop(startTime + 0.4);
  });
}
